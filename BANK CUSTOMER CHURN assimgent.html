# selection based on relevance to predict churn.
from scipy.stats import chi2_contingency

chi2_array, p_array = [], []
for column in categorical_feature:

    crosstab = pd.crosstab(df[column], df['Exited'])
    chi2, p, dof, expected = chi2_contingency( crosstab.values)
    chi2_array.append(chi2)
    p_array.append(p)

df_chi = pd.DataFrame({
    'feature':categorical_feature,
    'Chi-square': chi2,
    'p-value': p_array
})
df_chi.sort_values(by='Chi-square')
